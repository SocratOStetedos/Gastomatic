Extraits de code

Les patterns clés utilisés dans le projet

utils/api_client.py — Client HTTP réutilisable python 
import requests

BASE_URL = "https://jsonplaceholder.typicode.com"

def get_resource(endpoint: str) -> requests.Response:
    """Envoie une requête GET et retourne la réponse."""
    url = f"{BASE_URL}{endpoint}"
    return requests.get(url, timeout=10)

def create_resource(endpoint: str, payload: dict) -> requests.Response:
    """Envoie une requête POST avec un payload JSON."""
    url = f"{BASE_URL}{endpoint}"
    return requests.post(url, json=payload, timeout=10)

tests/test_posts.py — Tests avec classes et fixtures python 
class TestGetPosts:

    def test_get_single_post_has_required_fields(self, valid_post_id):
        """Vérifie que le JSON contient id, title, body, userId."""
        response = get_resource(f"/posts/{valid_post_id}")
        data = response.json()

        required_fields = ["id", "title", "body", "userId"]
        for field in required_fields:
            assert field in data, f"Champ manquant : '{field}'"
tests/test_errors.py — Tests paramétrés (1 test × N cas) python 
@pytest.mark.parametrize("endpoint", [
    "/posts/99999",
    "/users/99999",
    "/comments/99999",
])
def test_inexistant_resources_return_404(self, endpoint):
    """Ce test s'exécute 3 fois avec des valeurs différentes."""
    response = get_resource(endpoint)
    assert response.status_code == 404
